118 static bool _remove_events_with_arg(void * arg){
119  lwip_event_packet_t * first_packet = NULL;
120  lwip_event_packet_t * packet = NULL;
121 
122  if(!_async_queue){
123  return false;
124  }
125  //figure out which is the first packet so we can keep the order
126  while(!first_packet){
127  if(xQueueReceive(_async_queue, &first_packet, 0) != pdPASS){
128  return false;
129  }
130  //discard packet if matching
131  if((int)first_packet->arg == (int)arg){
132  free(first_packet);
133  first_packet = NULL;
134  //return first packet to the back of the queue
135  } else if(xQueueSend(_async_queue, &first_packet, portMAX_DELAY) != pdPASS){
136  return false;
137  }
138  }
139 
140  while(xQueuePeek(_async_queue, &packet, 0) == pdPASS && packet != first_packet){
141  if(xQueueReceive(_async_queue, &packet, 0) != pdPASS){
142  return false;
143  }
144  if((int)packet->arg == (int)arg){
145  free(packet);
146  packet = NULL;
147  } else if(xQueueSend(_async_queue, &packet, portMAX_DELAY) != pdPASS){
148  return false;
149  }
150  }
151  return true;
152 }

154 static void _handle_async_event(lwip_event_packet_t * e){
155  if(e->arg == NULL){
156  // do nothing when arg is NULL
157  //ets_printf("event arg == NULL: 0x%08x\n", e->recv.pcb);
158  } else if(e->event == LWIP_TCP_CLEAR){
159  _remove_events_with_arg(e->arg);
160  } else if(e->event == LWIP_TCP_RECV){
161  //ets_printf("-R: 0x%08x\n", e->recv.pcb);
162  AsyncClient::_s_recv(e->arg, e->recv.pcb, e->recv.pb, e->recv.err);
163  } else if(e->event == LWIP_TCP_FIN){
164  //ets_printf("-F: 0x%08x\n", e->fin.pcb);
165  AsyncClient::_s_fin(e->arg, e->fin.pcb, e->fin.err);
166  } else if(e->event == LWIP_TCP_SENT){
167  //ets_printf("-S: 0x%08x\n", e->sent.pcb);
168  AsyncClient::_s_sent(e->arg, e->sent.pcb, e->sent.len);
169  } else if(e->event == LWIP_TCP_POLL){
170  //ets_printf("-P: 0x%08x\n", e->poll.pcb);
171  AsyncClient::_s_poll(e->arg, e->poll.pcb);
172  } else if(e->event == LWIP_TCP_ERROR){
173  //ets_printf("-E: 0x%08x %d\n", e->arg, e->error.err);
174  AsyncClient::_s_error(e->arg, e->error.err);
175  } else if(e->event == LWIP_TCP_CONNECTED){
176  //ets_printf("C: 0x%08x 0x%08x %d\n", e->arg, e->connected.pcb, e->connected.err);
177  AsyncClient::_s_connected(e->arg, e->connected.pcb, e->connected.err);
178  } else if(e->event == LWIP_TCP_ACCEPT){
179  //ets_printf("A: 0x%08x 0x%08x\n", e->arg, e->accept.client);
180  AsyncServer::_s_accepted(e->arg, e->accept.client);
181  } else if(e->event == LWIP_TCP_DNS){
182  //ets_printf("D: 0x%08x %s = %s\n", e->arg, e->dns.name, ipaddr_ntoa(&e->dns.addr));
183  AsyncClient::_s_dns_found(e->dns.name, &e->dns.addr, e->arg);
184  }
185  free((void*)(e));
186 }
