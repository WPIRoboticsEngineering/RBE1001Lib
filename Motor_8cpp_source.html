213 void Motor::loop() {
214  nowEncoder = encoder->getCount();
215  if (closedLoopControl) {
216  //portEXIT_CRITICAL(&mmux);
217  if (mode == VELOCITY_MODE) {
218  Setpoint += milisecondPosIncrementForVelocity;
219  } else {
220  unitDuration = getInterpolationUnitIncrement();
221  if (unitDuration < 1) {
222  float setpointDiff = endSetpoint - startSetpoint;
223  float newSetpoint = startSetpoint
224  + (setpointDiff * unitDuration);
225  Setpoint = newSetpoint;
226  } else {
227  // If there is no interpoation to perform, set the setpoint to the end state
228  Setpoint = endSetpoint;
229  }
230  }
231  float controlErr = Setpoint - nowEncoder;
232  // shrink old values out of the sum
233  runntingITerm = runntingITerm * ((I_TERM_SIZE - 1.0) / I_TERM_SIZE);
234  // running sum of error
235  runntingITerm += controlErr;
236 
237  currentEffort = controlErr * kP + ((runntingITerm / I_TERM_SIZE) * kI);
238 
239  //portEXIT_CRITICAL(&mmux);
240  }
241 
242  interruptCountForVelocity++;
243  if (interruptCountForVelocity == 50) {
244  interruptCountForVelocity = 0;
245  float err = prevousCount - nowEncoder;
246  cachedSpeed = err / (0.05); // ticks per second
247  prevousCount = nowEncoder;
248  }
249  SetEffortLocal(currentEffort);
250 
251 }
