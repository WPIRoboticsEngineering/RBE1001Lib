208 void Motor::loop() {
209  nowEncoder = encoder->getCount();
210  if (closedLoopControl) {
211  //portEXIT_CRITICAL(&mmux);
212  if (mode == VELOCITY_MODE) {
213  Setpoint += milisecondPosIncrementForVelocity;
214  } else {
215  unitDuration = getInterpolationUnitIncrement();
216  if (unitDuration < 1) {
217  float setpointDiff = endSetpoint - startSetpoint;
218  float newSetpoint = startSetpoint
219  + (setpointDiff * unitDuration);
220  Setpoint = newSetpoint;
221  } else {
222  // If there is no interpoation to perform, set the setpoint to the end state
223  Setpoint = endSetpoint;
224  }
225  }
226  float controlErr = Setpoint - nowEncoder;
227  // shrink old values out of the sum
228  runntingITerm = runntingITerm * ((I_TERM_SIZE - 1.0) / I_TERM_SIZE);
229  // running sum of error
230  runntingITerm += controlErr;
231 
232  currentEffort = controlErr * kP + ((runntingITerm / I_TERM_SIZE) * kI);
233 
234  //portEXIT_CRITICAL(&mmux);
235  }
236 
237  interruptCountForVelocity++;
238  if (interruptCountForVelocity == 50) {
239  interruptCountForVelocity = 0;
240  float err = prevousCount - nowEncoder;
241  cachedSpeed = err / (0.05); // ticks per second
242  prevousCount = nowEncoder;
243  }
244  SetEffortLocal(currentEffort);
245 
246 }
