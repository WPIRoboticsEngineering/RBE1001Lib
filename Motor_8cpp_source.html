28 float Motor::getInterpolationUnitIncrement() {
29  float interpElapsed = (float) (millis() - startTime);
30  if (interpElapsed < duration && duration > 0) {
31  // linear elapsed duration
32  unitDuration = interpElapsed / duration;
33  if (mode == SINUSOIDAL_INTERPOLATION) {
34  // sinusoidal ramp up and ramp down
35  float sinPortion = (cos(-PI * unitDuration) / 2) + 0.5;
36  unitDuration = 1 - sinPortion;
37  }
38  if (mode == BEZIER) {
39  if (unitDuration > 0 && unitDuration < 1) {
40  float t = unitDuration;
41  float P0 = 0;
42  float P1 = BEZIER_P0;
43  float P2 = BEZIER_P1;
44  float P3 = 1;
45  unitDuration = pow((1 - t), 3) * P0
46  + 3 * t * pow((1 - t), 2) * P1
47  + 3 * pow(t, 2) * (1 - t) * P2 + pow(t, 3) * P3;
48  }
49  }
50  if (mode == TRAPEZOIDAL) {
51  float lengthOfLinearMode = duration - (TRAPEZOIDAL_time * 2);
52  float unitLienear = lengthOfLinearMode / duration;
53  float unitRamp = ((float) TRAPEZOIDAL_time) / duration;
54  float unitStartRampDown = unitLienear + unitRamp;
55  if (unitDuration < unitRamp) {
56  // ramp up
57  // range from 1 to 0.5
58  float increment = 1 - (unitDuration) / (unitRamp * 2);
59  // range 0 to 1
60  float sinPortion = 1 + cos(-PI * increment);
61  unitDuration = sinPortion * unitRamp;
62  } else if (unitDuration > unitRamp
63  && unitDuration < unitStartRampDown) {
64  // constant speed
65 
66  } else if (unitDuration > unitStartRampDown) {
67  float increment = (unitDuration - unitStartRampDown)
68  / (unitRamp * 2) + 0.5;
69  float sinPortion = 0.5 - ((cos(-PI * increment) / 2) + 0.5);
70  unitDuration = (sinPortion * 2) * unitRamp + unitStartRampDown;
71  }
72  }
73  return unitDuration;
74  }
75  return 1;
76 }

179 bool Motor::isMotorDoneWithMove() {
180 // First wait for the interpolation to finish
181  if (getInterpolationUnitIncrement() < 1) {
182  ESP_LOGD(TAG,"Move Interpolation Remaining: %.4f",getInterpolationUnitIncrement());
183  return false;
184  }
185  float distanceToGo = fabs(
186  (setpoint * TICKS_TO_DEGREES) - getCurrentDegrees());
187 
188  if (distanceToGo > 0.75) { // more than 1 degree from target
189  ESP_LOGD(TAG,"Move Remaining: %.4f",distanceToGo);
190  return false;
191  }
192 // wait for the velocity to be below 10deg/sec

195  ESP_LOGD(TAG,"Move Speed: %.4f",getDegreesPerSecond());
196  return false;
197  }
198 // All moving checks came back passed!

288 void Motor::loop() {
289  nowEncoder = encoder->getCount();
290  if (closedLoopControl) {
291  //portEXIT_CRITICAL(&mmux);
292  if (mode == VELOCITY_MODE) {
293  if (fabs(currentEffort) < 0.95) // stall detection
294  setpoint += milisecondPosIncrementForVelocity;
295  } else {
296  unitDuration = getInterpolationUnitIncrement();
297  if (unitDuration < 1) {
298  float setpointDiff = endSetpoint - startSetpoint;
299  float newSetpoint = startSetpoint
300  + (setpointDiff * unitDuration);
301  setpoint = newSetpoint;
302  } else {
303  // If there is no interpoation to perform, set the setpoint to the end state
304  setpoint = endSetpoint;
305  }
306  }
307  float controlErr = setpoint - nowEncoder;
308 
309  if (getInterpolationUnitIncrement() < 1) {
310  // no i term during interpolation
311  runntingITerm = 0;
312  } else {
313  // shrink old values out of the sum
314  runntingITerm = runntingITerm * ((I_TERM_SIZE - 1.0) / I_TERM_SIZE);
315  // running sum of error
316  runntingITerm += controlErr;
317  }
318 
319  currentEffort =
320  -(controlErr * kP + ((runntingITerm / I_TERM_SIZE) * kI));
321 
322  //portEXIT_CRITICAL(&mmux);
323  }
324 
325  else
326  {
327  if(targetEffort > currentEffort + DELTA_EFFORT)
328  currentEffort += DELTA_EFFORT;
329  else if(targetEffort < currentEffort - DELTA_EFFORT)
330  currentEffort -= DELTA_EFFORT;
331  else currentEffort = targetEffort;
332  }
333 
334 
335  interruptCountForVelocity++;
336  if (interruptCountForVelocity == 50) {
337  interruptCountForVelocity = 0;
338  float err = prevousCount - nowEncoder;
339  cachedSpeed = err / (0.05); // ticks per second
340  prevousCount = nowEncoder;
341  }
342 // invert the effort so that the set speed and set effort match
343  setEffortLocal(currentEffort);
