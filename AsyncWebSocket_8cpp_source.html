54 size_t webSocketSendFrame(AsyncClient *client, bool final, uint8_t opcode, bool mask, uint8_t *data, size_t len){
55  if(!client->canSend())
56  return 0;
57  size_t space = client->space();
58  if(space < 2)
59  return 0;
60  uint8_t mbuf[4] = {0,0,0,0};
61  uint8_t headLen = 2;
62  if(len && mask){
63  headLen += 4;
64  mbuf[0] = rand() % 0xFF;
65  mbuf[1] = rand() % 0xFF;
66  mbuf[2] = rand() % 0xFF;
67  mbuf[3] = rand() % 0xFF;
68  }
69  if(len > 125)
70  headLen += 2;
71  if(space < headLen)
72  return 0;
73  space -= headLen;
74 
75  if(len > space) len = space;
76 
77  uint8_t *buf = (uint8_t*)malloc(headLen);
78  if(buf == NULL){
79  //os_printf("could not malloc %u bytes for frame header\n", headLen);
80  return 0;
81  }
82 
83  buf[0] = opcode & 0x0F;
84  if(final)
85  buf[0] |= 0x80;
86  if(len < 126)
87  buf[1] = len & 0x7F;
88  else {
89  buf[1] = 126;
90  buf[2] = (uint8_t)((len >> 8) & 0xFF);
91  buf[3] = (uint8_t)(len & 0xFF);
92  }
93  if(len && mask){
94  buf[1] |= 0x80;
95  memcpy(buf + (headLen - 4), mbuf, 4);
96  }
97  if(client->add((const char *)buf, headLen) != headLen){
98  //os_printf("error adding %lu header bytes\n", headLen);
99  free(buf);
100  return 0;
101  }
102  free(buf);
103 
104  if(len){
105  if(len && mask){
106  size_t i;
107  for(i=0;i<len;i++)
108  data[i] = data[i] ^ mbuf[i%4];
109  }
110  if(client->add((const char *)data, len) != len){
111  //os_printf("error adding %lu data bytes\n", len);
112  return 0;
113  }
114  }
115  if(!client->send()){
116  //os_printf("error sending frame: %lu\n", headLen+len);
117  return 0;
118  }
119  return len;
120 }

621 void AsyncWebSocketClient::_onData(void *pbuf, size_t plen){
622  _lastMessageTime = millis();
623  uint8_t *data = (uint8_t*)pbuf;
624  while(plen > 0){
625  if(!_pstate){
626  const uint8_t *fdata = data;
627  _pinfo.index = 0;
628  _pinfo.final = (fdata[0] & 0x80) != 0;
629  _pinfo.opcode = fdata[0] & 0x0F;
630  _pinfo.masked = (fdata[1] & 0x80) != 0;
631  _pinfo.len = fdata[1] & 0x7F;
632  data += 2;
633  plen -= 2;
634  if(_pinfo.len == 126){
635  _pinfo.len = fdata[3] | (uint16_t)(fdata[2]) << 8;
636  data += 2;
637  plen -= 2;
638  } else if(_pinfo.len == 127){
639  _pinfo.len = fdata[9] | (uint16_t)(fdata[8]) << 8 | (uint32_t)(fdata[7]) << 16 | (uint32_t)(fdata[6]) << 24 | (uint64_t)(fdata[5]) << 32 | (uint64_t)(fdata[4]) << 40 | (uint64_t)(fdata[3]) << 48 | (uint64_t)(fdata[2]) << 56;
640  data += 8;
641  plen -= 8;
642  }
643 
644  if(_pinfo.masked){
645  memcpy(_pinfo.mask, data, 4);
646  data += 4;
647  plen -= 4;
648  }
649  }
650 
651  const size_t datalen = std::min((size_t)(_pinfo.len - _pinfo.index), plen);
652  const auto datalast = data[datalen];
653 
654  if(_pinfo.masked){
655  for(size_t i=0;i<datalen;i++)
656  data[i] ^= _pinfo.mask[(_pinfo.index+i)%4];
657  }
658 
659  if((datalen + _pinfo.index) < _pinfo.len){
660  _pstate = 1;
661 
662  if(_pinfo.index == 0){
663  if(_pinfo.opcode){
664  _pinfo.message_opcode = _pinfo.opcode;
665  _pinfo.num = 0;
666  } else _pinfo.num += 1;
667  }
668  _server->_handleEvent(this, WS_EVT_DATA, (void *)&_pinfo, (uint8_t*)data, datalen);
669 
670  _pinfo.index += datalen;
671  } else if((datalen + _pinfo.index) == _pinfo.len){
672  _pstate = 0;
673  if(_pinfo.opcode == WS_DISCONNECT){
674  if(datalen){
675  uint16_t reasonCode = (uint16_t)(data[0] << 8) + data[1];
676  char * reasonString = (char*)(data+2);
677  if(reasonCode > 1001){
678  _server->_handleEvent(this, WS_EVT_ERROR, (void *)&reasonCode, (uint8_t*)reasonString, strlen(reasonString));
679  }
680  }
681  if(_status == WS_DISCONNECTING){
682  _status = WS_DISCONNECTED;
683  _client->close(true);
684  } else {
685  _status = WS_DISCONNECTING;
686  _client->ackLater();
687  _queueControl(new AsyncWebSocketControl(WS_DISCONNECT, data, datalen));
688  }
689  } else if(_pinfo.opcode == WS_PING){
690  _queueControl(new AsyncWebSocketControl(WS_PONG, data, datalen));
691  } else if(_pinfo.opcode == WS_PONG){
692  if(datalen != AWSC_PING_PAYLOAD_LEN || memcmp(AWSC_PING_PAYLOAD, data, AWSC_PING_PAYLOAD_LEN) != 0)
693  _server->_handleEvent(this, WS_EVT_PONG, NULL, data, datalen);
